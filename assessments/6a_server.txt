from flask import Flask, request, render_template
from EmotionDetection.EmotionDetection import emotion_detector

app = Flask(__name__)

@app.route('/')
def home():
    return render_template('index.html')


@app.route('/emotionDetector')
def emotion_detector_route():
    text_to_analyze = request.args.get('textToAnalyze')


    if not text_to_analyze:
        return "Invalid input! Please provide text to analyze."

    # Get result from emotion detector (dictionary)
    result = emotion_detector(text_to_analyze)

    # Extract emotions (exclude dominant_emotion)
    emotions = {
        k: v for k, v in result.items()
        if k != "dominant_emotion"
    }

    dominant_emotion = result["dominant_emotion"]

    # Format emotion string exactly as IBM requires
    emotion_items = [
        f"'{k}': {v:.9f}" for k, v in emotions.items()
    ]

    emotions_str = ", ".join(emotion_items[:-1]) + " and " + emotion_items[-1]

    # Final required sentence format
    return (
        f"For the given statement, the system response is {emotions_str}. "
        f"The dominant emotion is {dominant_emotion}."
    )


if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)
